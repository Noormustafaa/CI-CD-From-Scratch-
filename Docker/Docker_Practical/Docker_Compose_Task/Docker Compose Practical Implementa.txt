Step 1: Python Application Code (app.py)
Sab se pehle apne folder mein app.py naam ki file banayein aur usme ye code likhein:

import time
import redis
from flask import Flask

app = Flask(__name__)
# 'redis' yahan host name hai, jo Docker Compose network ke zariye automatic connect hoga
cache = redis.Redis(host='redis', port=6379)

def get_hit_count():
    retries = 5
    while True:
        try:
            return cache.incr('hits')
        except redis.exceptions.ConnectionError as exc:
            if retries == 0:
                raise exc
            retries -= 1
            time.sleep(0.5)

@app.route('/')
def hello():
    count = get_hit_count()
    return f'Hello World! I have been seen {count} times.\n'

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000, debug=True)
